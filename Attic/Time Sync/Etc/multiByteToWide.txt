        static wchar_t* multiByteToWide(char* string) {
            size_t length = strlen(string);
            wchar_t* wstring = new wchar_t[length + 1];

            int result = MultiByteToWideChar(CP_UTF8, MB_ERR_INVALID_CHARS,
                string, -1, wstring, length + 1);

            if (result == 0) {
                delete[] wstring;
                return NULL;
            }

            return wstring;
        }
